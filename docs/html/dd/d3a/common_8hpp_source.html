187 template<typename Integral, typename = std::enable_if_t<std::is_integral_v<Integral>, std::nullptr_t>>
188 std::vector<Integral> to_integral(const types::argument_value_type& value)
189 {
190 std::vector<Integral> numvals;
191 switch (value.index())
192 {
193 case 1:
194 if constexpr (std::is_signed_v<Integral>)
195 {
196 auto numval = std::stoll(std::get<1>(value));
197 numvals.push_back(static_cast<Integral>(numval));
198 }
199 else
200 {
201 auto numval = std::stoull(std::get<1>(value));
202 numvals.push_back(static_cast<Integral>(numval));
203 }
204 break;
205 case 2:
206 const auto& strvals = std::get<2>(value);
207 numvals.reserve(strvals.size());
208 for(const auto& strval : strvals)
209 {
210 if constexpr (std::is_signed_v<Integral>)
211 {
212 auto numval = std::stoll(strval);
213 numvals.push_back(static_cast<Integral>(numval));
214 }
215 else
216 {
217 auto numval = std::stoull(strval);
218 numvals.push_back(static_cast<Integral>(numval));
219 }
220 }
221 }
222 return numvals;
223 }
